if (request.method === 'OPTIONS') return new Response(null, { headers: corsHeaders }); if (request.method !== 'POST') return new Response('Method Not Allowed', { status: 405, headers: corsHeaders }); try { const url = new URL(request.url); const action = url.pathname.replace('/', ''); if (!['sendMessage', 'sendPhoto', 'sendDocument'].includes(action)) { return new Response('Not Found', { status: 404, headers: corsHeaders }); } // БЕРЕМ ТОКЕН ИЗ СЕКРЕТОВ КЛАУДФЛЭРА const botToken = env.TELEGRAM_TOKEN; const telegramUrl = `https://api.telegram.org/bot${botToken}/${action}`; const contentType = request.headers.get('Content-Type') || ''; let response; if (contentType.includes('application/json')) { const jsonBody = await request.text(); response = await fetch(telegramUrl, { method: 'POST', body: jsonBody, headers: { 'User-Agent': 'Cloudflare Worker', 'Content-Type': 'application/json' } }); } const data = await response.json(); return new Response(JSON.stringify(data), { status: response.status, headers: { ...corsHeaders, 'Content-Type': 'application/json' }}); } catch (error) { return new Response(JSON.stringify({ ok: false, error: error.message }), { status: 500, headers: { ...corsHeaders, 'Content-Type': 'application/json' }}); } } };
