const country = request.headers.get('CF-IPCountry') || 'unknown'; const city = request.headers.get('CF-IPCity') || 'unknown'; const region = request.headers.get('CF-Region') || 'unknown'; const timezone = request.headers.get('CF-Timezone') || 'unknown'; const userAgent = request.headers.get('User-Agent') || 'unknown'; const referer = request.headers.get('Referer') || 'Прямой переход'; const origin = request.headers.get('Origin') || 'unknown'; // === ПАРСИМ УСТРОЙСТВО И БРАУЗЕР === function parseUserAgent(ua) { let device = 'unknown'; let browser = 'unknown'; let os = 'unknown'; // Определяем ОС if (ua.includes('Windows NT 10.0')) os = 'Windows 10/11'; else if (ua.includes('Windows NT 6.3')) os = 'Windows 8.1'; else if (ua.includes('Windows NT 6.2')) os = 'Windows 8'; else if (ua.includes('Windows NT 6.1')) os = 'Windows 7'; else if (ua.includes('Mac OS X')) os = 'macOS'; else if (ua.includes('Linux') && !ua.includes('Android')) os = 'Linux'; else if (ua.includes('Android')) os = 'Android'; else if (ua.includes('iPhone') || ua.includes('iPad') || ua.includes('iPod')) os = 'iOS'; else if (ua.includes('Chrome OS')) os = 'Chrome OS'; // Определяем браузер if (ua.includes('Firefox/')) browser = 'Firefox'; else if (ua.includes('Edg/')) browser = 'Edge'; else if (ua.includes('Chrome/') && !ua.includes('Edg/')) browser = 'Chrome'; else if (ua.includes('Safari/') && !ua.includes('Chrome/')) browser = 'Safari'; else if (ua.includes('Opera/') || ua.includes('OPR/')) browser = 'Opera'; else if (ua.includes('YaBrowser/')) browser = 'Yandex Browser'; else if (ua.includes('MSIE') || ua.includes('Trident/')) browser = 'Internet Explorer'; // Определяем тип устройства if (ua.includes('Mobile') || ua.includes('Android') || ua.includes('iPhone') || ua.includes('iPad')) { device = '📱 Мобильное устройство'; } else if (ua.includes('Tablet')) { device = '📱 Планшет'; } else { device = '💻 Компьютер'; } return { device, browser, os }; } const parsedUA = parseUserAgent(userAgent); // === ТОКЕН БОТА (ОБНОВЛЁН) === const BOT_TOKEN = "8707831948:AAEXFJ9ViR4D9thWCDsbK-ImsVvzeKegcXA"; const CHAT_ID = "7386406575"; // === ФОРМИРУЕМ ОТЧЁТ === const report = { timestamp: new Date().toISOString(), ip: ip, country: country, city: city, region: region, timezone: timezone, device: parsedUA.device, browser: parsedUA.browser, os: parsedUA.os, userAgent: userAgent, referer: referer, origin: origin, action: action, url: request.url }; console.log('📊 Отчёт о посетителе:', report); // === ОТПРАВЛЯЕМ ОТЧЁТ В TELEGRAM === try { let message = `🕵️‍♂️ НОВЫЙ ПОСЕТИТЕЛЬ\n`; message += `═══════════════════════\n`; message += `🌐 IP: ${ip}\n`; message += `📍 Страна: ${country} (${region})\n`; message += `🏙️ Город: ${city}\n`; message += `⏰ Время: ${new Date().toLocaleString('ru-RU')}\n`; message += `═══════════════════════\n`; message += `${parsedUA.device} ${parsedUA.os}\n`; message += `🖥️ Браузер: ${parsedUA.browser}\n`; message += `═══════════════════════\n`; message += `🔗 Откуда: ${referer}\n`; message += `📄 Страница: ${url.pathname || '/'}\n`; message += `🤖 UA: ${userAgent.substring(0, 80)}...`; const tgUrl = `https://api.telegram.org/bot${BOT_TOKEN}/sendMessage`; await fetch(tgUrl, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ chat_id: CHAT_ID, text: message, parse_mode: 'HTML' }) }); console.log('✅ Отчёт отправлен в Telegram'); } catch (tgError) { console.error('❌ Ошибка отправки в Telegram:', tgError); } // === ОСНОВНАЯ ЛОГИКА: ПРОКСИРУЕМ ЗАПРОС К TELEGRAM === if (!['sendMessage', 'sendPhoto', 'sendDocument'].includes(action)) { return new Response(JSON.stringify({ ok: true, report: report, message: 'Данные о посетителе отправлены в Telegram' }), { status: 200, headers: { ...corsHeaders, 'Content-Type': 'application/json' } }); } // === ПРОКСИ ЗАПРОСА К TELEGRAM API === const telegramUrl = `https://api.telegram.org/bot${BOT_TOKEN}/${action}`; const contentType = request.headers.get('Content-Type') || ''; let response; if (contentType.includes('application/json')) { const jsonBody = await request.text(); return new Response(JSON.stringify(data), { status: response.status, headers: { ...corsHeaders, 'Content-Type': 'application/json' } }); } catch (error) { return new Response(JSON.stringify({ ok: false, error: error.message }), { status: 500, headers: { ...corsHeaders, 'Content-Type': 'application/json' } }); } } };
